class Ball {
 
int size = 50;
float xpos;
float ypos;
float xspeed;
float yspeed;
color c;
int xdirection = 1;
int ydirection = 1;
 
Ball (color c_, float xp, float yp,float xs, float ys){
 c = c_;
 xpos = xp;
 ypos = yp;
 xspeed = xs;
 yspeed = ys;
 
}
 
void draw(){
 ellipseMode(CENTER_DIAMETER);
 noStroke();
 fill (c);
 smooth();
 ellipse(xpos,ypos,size,size);
}

 xdirection *=-1;}
