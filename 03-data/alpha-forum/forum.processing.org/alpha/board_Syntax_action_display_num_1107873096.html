BImage img;
 
void setup(){
size(200, 200);
  img = loadImage("http://iwin.nws.noaa.gov/iwin/images/ecir.jpg");
}
 
void loop(){
  background(0);
  image(img, mouseX, mouseY);
}
 
-------------------------------
 
 
BImage img;
int top;
 
void setup(){
size(200, 200);
top=0;
  img = loadImage("http://iwin.nws.noaa.gov/iwin/images/ecir.jpg");
}
 
void loop(){
  background(0);
  image(img, 0, top);
top=(top+1)%height;
}
