 size(300,600);

I believe you cannot load the String befor the setup methode
 
Code:

String [] lines;
keyword [] keywords;
 
void setup(){
  size(200,400);
  lines = loadStrings("list.txt");
  keywords = new keyword[lines.length];
 
  BFont keywordFont;  
  keywordFont = loadFont("Tahoma.vlw");  
  textFont(keywordFont, 22);  
 
  // setup the keyword array
  for(int i=0; i<lines.length; i++) {
    keywords[i] = new keyword(lines[i],    (int)random(30,300), 15);
  }
  framerate(1);
}
