please take a look at the following depth effect that works okay with QUAD_STRIP:
http://www.chronotext.org/processing/depth_002
 
Code:

float fovy, aspect;
float elevation, azimuth, distance;
 
void setup()
{
  size(300, 300);
  framerate(20.0);
  background(0);
 
  fovy = 60.0;
  aspect = (float) width / (float) height;
 
  distance = 220.0;
  elevation = radians(300.0);
  azimuth = radians(0.0);
}
 
void loop()
{
  beginCamera();
  perspective(fovy, aspect, 1.0, 1000.0);
  translate(0.0, 0.0, -distance);
  rotateX(-elevation);
  rotateZ(-azimuth);
  endCamera();
 
  ring(0.0, 0.0, 0.0, 100.0, 30.0, radians(12.0));  // the last param controls L.O.D
 
  azimuth += radians(2.0);
}
 
void ring(float x, float y, float z, float r, float h, float dd)
{
  boolean b = false;
  float xx, yy;
 
  noStroke();
  beginShape(QUAD_STRIP);
  for (float d = 0; d < TWO_PI; d += dd)
  {
    b = !b;
    xx = x + cos(d) * r;
    yy = y - sin(d) * r;
    fill(255, 0, 0, 153.0 + sin(PI + d + azimuth) * 102.0);  // depth effect...
    vertex(xx, yy, b ? z : z + h);  
    vertex(xx, yy, b ? z + h : z);  
   }
  endShape();
}

okay, finally i found a workaround for vertex coloring and texture mapping:
http://proce55ing.net/discourse/yabb/board_Proce55ing_software__bugs_action_display_num_1054456405.html
 
the bottom line: vertex coloring (including alpha coloring) can work for texture mapping, but only if hint(SMOOTH_IMAGES) is turned on and the image used is of format ALPHA
 
ligths() can affect images too, the same restrictions as above are applied...
