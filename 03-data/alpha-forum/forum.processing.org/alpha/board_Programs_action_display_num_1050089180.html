class FastBlend {

  FastBlend() {
    float blend=0;
    for(int i=0; i<21; i++) {

  int blend(int c1, float b1, int c2, float b2) {

    return min(t1[(c1>>16)&0xff] +t2[(c2>>16)&0xff],0xff)<<16 |
