//Here you define the rate  
float rate = 2;  
 
//x this is the initial value
float init_val = 0.1  ;
 
//calculate for X0  
values[0] = (rate * init_val) * ( 1 - init_val);
 
//print X0 value
println(values[0]);
 
//start the loop and calculate for i= 1 to n  
for( int i = 1; i < n; i++ ) {
//print i
println(i);
 
//this is the logistic equation
 values[i] = (rate * values[i-1]) * (1 - values[i-1]);
