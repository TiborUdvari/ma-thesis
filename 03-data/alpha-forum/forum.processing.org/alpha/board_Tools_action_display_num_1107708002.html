Yeah. I think I might need a tool like that at some point.
Code:

BImage file;
void setup(){
file = loadImage ("d.gif");
size(file.width, file.height);
}
void draw(){
for (int i = 0; i < file.pixels.length; i++){
whiter(file.pixels[i],i);
}
background(file);
save("white.tif");
println("done");
}
//whiting out function
void whiter(color w, int p){
/*
we've got three 255 values in a color
add those numbers together and colors near white
should be from the max(765) to say 600?
play around with the value and add stuff
to the red, green and blue values to catch
other stuff.
*/
if (red(w)+blue(w)+green(w) > 600){
file.pixels[p] = #FFFFFF; //or change this to your transparency color for a .gif
}
}
