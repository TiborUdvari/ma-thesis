int i,j,k,l;
void setup() {
size(500,500);
background(255);
smooth();
strokeWeight(1);
}
void draw() {
stroke(0);
fill(0);
beginShape(POLYGON);
for(i =0; i<11; i++) {

curveVertex(50*i,j);

vertex(50*(i-2),240);

}

can't help but here's a pretty accident courtesy of the strokeWeight command:
 
Code:

 int i,j,k,l,sw;
void setup() {
size(500,500);
background(255);
smooth();
}
void loop() {
background(255);
sw++;
stroke(0,25);
beginShape(POLYGON);
for(i =0; i<11; i++) {
j = (i%2)*50 + 200;
curveVertex(50*i,j);
}
vertex(50*(i-2),240);
for(k = (i-1); k>-1; k--) {
int l = (k%2)*50 + 190;
curveVertex(50*k, l);
}
endShape();
strokeWeight(sw);
}
